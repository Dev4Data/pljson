/* Outputting unescaped values */

set serveroutput on;
declare 
  obj json;
begin
  obj := json();
  obj.put('foo1', json_value('function() { a = 12;}'));
  obj.put('foo2', json_value('function() { a = 12;}', false));
  obj.print; --beware - the output does not conform to the JSON standard now.
end;
/
